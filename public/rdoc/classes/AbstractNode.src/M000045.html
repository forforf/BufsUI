  def self.create(node)
    ##TODO: move name_space stuff to initialize method
    #name_space = nil
    #name_space = node.class.name_space if node.class.name_space
    #@full_name_space = "#{node.class}-#{name_space}"
    case node.class.to_s
      when 'BufsInfoDoc'
        return DBDocNode.new(node)
      when 'BufsFileSystem'
        return FileSystemDocNode.new(node)
      else
        raise "Abstract Node could not create node, unknown Node Type of: #{node.class.to_s.inspect}"
    end
  end
